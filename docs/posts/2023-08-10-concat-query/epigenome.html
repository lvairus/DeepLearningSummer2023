def get_chr_ints_list(chr_num, chr_len,):
    intervals = []
    
    # determine number of tiles
    num_tiles = chr_len // L2
    if L2*(num_tiles)+L4 < chr_len:
        num_tiles += 1

    # create intervals
    for i in range(num_tiles):
        start = L2 * i
        end = L2 * (i+2)
        interval = f'chr{chr_num}_{start}_{end}'
        intervals.append(interval)
            
    return intervals
